410  /**
411   * Inspects all types with {@link Requirement} annotations and adds
412   * {@link Inject} and {@link Hinted} annotations on each {@link
413   * Requirement}-annotated field found.
414   *
415   * <p>This has the net effect of turning a Plexus injection point
416   * into a CDI injection point.</p>
417   *
418   * @param <X> the type being processed
419   *
420   * @param event the {@link ProcessAnnotatedType} event describing
421   * the type to process; may be {@code null} in which case no action
422   * will be taken
423   *
424   * @see Requirement
425   */
426  private final <X> void processPlexusRequirementAnnotatedMembers(@Observes @WithAnnotations(Requirement.class) final ProcessAnnotatedType<X> event) {
427    if (event != null) {
428      final AnnotatedType<X> type = event.getAnnotatedType();
429      if (type != null && type.getConstructors().stream().noneMatch(c -> c.isAnnotationPresent(Inject.class))) {
430        event.configureAnnotatedType()
431          .filterFields(f -> {
432              return f != null && f.isAnnotationPresent(Requirement.class) && !f.isAnnotationPresent(Inject.class);
433            })
434          .forEach(fc -> {
435              fc.add(InjectLiteral.INSTANCE);
436              final AnnotatedField<?> f = fc.getAnnotated();
437              if (f != null && !f.isAnnotationPresent(Hinted.class)) {
438                final Requirement requirement = f.getAnnotation(Requirement.class);
439                if (requirement != null) {
440                  final String hint = requirement.hint();
441                  if (hint != null && !hint.trim().isEmpty()) {
442                    fc.add(Hinted.Literal.of(hint));
443                  }
444                }
445              }
446            });
447      }
448    }
449  }

451  /**
452   * Inspects all types annotated with {@link Component} to see if
453   * they have a designated {@linkplain Component#role() role}, and,
454   * if so, adds an appropriate {@link Typed} annotation to them as
455   * well as a {@link Hinted} annotation, if necessary.
456   *
457   * @param <X> the type being processed
458   *
459   * @param event the {@link ProcessAnnotatedType} event describing
460   * the type to process; may be {@code null} in which case no action
461   * will be taken
462   *
463   * @see Component
464   */
465  private final <X> void processPlexusComponentAnnotatedTypes(@Observes @WithAnnotations(Component.class) final ProcessAnnotatedType<X> event) {
466    if (event != null) {
467      final AnnotatedType<X> type = event.getAnnotatedType();
468      if (type != null && !type.isAnnotationPresent(Typed.class)) {
469        final Component component = type.getAnnotation(Component.class);
470        if (component != null) {
471          final AnnotatedTypeConfigurator<X> atc = event.configureAnnotatedType();
472          if (atc != null) {
473            final Class<?> role = component.role();
474            if (role != null && !role.equals(Object.class)) {
475              atc.add(Typed.Literal.of(new Class<?>[] { role, Object.class }));
476            }
477            final String hint = component.hint();
478            if (hint != null && !hint.isEmpty() && !type.isAnnotationPresent(Hinted.class)) {
479              atc.add(Hinted.Literal.of(hint));
480            }
481          }
482        }
483      }
484    }

487  /**
488   * Returns a {@link Properties} object containing the <a
489   * href="https://maven.apache.org/shared/maven-archiver/#pom-properties-content"
490   * target="_parent">contents of a {@code pom.properties} file as
491   * commonly installed in {@code .jar} files by Maven</a>.
492   *
493   * <p>This method may return {@code null}.</p>
494   *
495   * <p>This method is not {@code final} due to CDI specification
496   * restrictions only.</p>
497   *
498   * @param location the location of the classpath element (a {@code
499   * .jar} file, or, perhaps, a directory); may be {@code null} in
500   * which case {@code null} will be returned
501   *
502   * @return a {@link Properties} object, or {@code null}
503   *
504   * @exception IOException if an input/output error occurs
505   *
506   * @exception URISyntaxException if there was a problem constructing
507   * a {@link URI}
508   */
509  public Properties getGroupArtifactVersionCoordinates(final URL location) throws IOException, URISyntaxException {
510    Properties returnValue = null;
511    if (location != null) {
512      final String path = location.getPath();
513
514      URL beanArchiveRoot = null;
515
516      String scheme = location.getProtocol();
517      if ("jar".equalsIgnoreCase(scheme)) {
518        beanArchiveRoot = new URL(location, "/"); // no jar entry, no matter what was passed in
519      } else if (path != null && path.endsWith(".jar")) {
520        beanArchiveRoot = new URL(new URL("jar", "", -1, location.toString() + "!/"), "/"); // normalize to jar URL with no jar entry
521      } else {
522        beanArchiveRoot = location; // no idea
523      }
524      assert beanArchiveRoot != null;
525
526      synchronized (this.coordinates) {
527        returnValue = this.coordinates.get(beanArchiveRoot.toURI());
528        if (returnValue == null) {
529
530          scheme = beanArchiveRoot.getProtocol();
531          if ("jar".equalsIgnoreCase(scheme)) {
532            final JarURLConnection urlConnection = (JarURLConnection)beanArchiveRoot.openConnection();
533            if (urlConnection != null) {
534              try (final JarFile jarFile = urlConnection.getJarFile()) {
535                if (jarFile != null) {
536                  returnValue = jarFile.stream()
537                    .filter(e -> {
538                        final String name = e.getName();
539                        return name != null && name.startsWith("META-INF/maven/") && name.endsWith("/pom.properties");
540                      })
541                    .findAny() // there should be only one
542                    .map(jarEntry -> {
543                        Properties p = null;
544                        try (final InputStream inputStream = new BufferedInputStream(jarFile.getInputStream(jarEntry))) {
545                          if (inputStream != null) {
546                            p = new Properties();
547                            p.load(inputStream);
548                          }
549                        } catch (final IOException ioException) {
550                          throw new RuntimeException(ioException);
551                        }
552                        return p;
553                      })
554                    .orElse(null);
555                }
556              }
557            }
558          } else if ("file".equalsIgnoreCase(scheme)) {
559            File root = new File(beanArchiveRoot.toURI());
560            while (root != null) {
561              if (root.isDirectory()) {
562                final File metaInfMaven = new File(root, "META-INF/maven");
563                if (metaInfMaven.isDirectory()) {
564
565                  File scanMe = metaInfMaven;
566                  while (scanMe != null && scanMe.isDirectory()) {
567
568                    final File[] subDirectories = scanMe.listFiles(f -> {
569                        if (f == null || !f.isDirectory()) {
570                          return false;
571                        }
572                        final String name = f.getName();
573                        return name != null && !name.startsWith(".");
574                      });
575
576                    if (subDirectories == null || subDirectories.length <= 0) {
577                      // We are looking for pom.properties in a
578                      // directory that has no further subdirectories.
579                      // We've found it.
580                      final File pomPropertiesFile = new File(scanMe, "pom.properties");
581                      scanMe = null;
582                      if (pomPropertiesFile.exists() && !pomPropertiesFile.isDirectory()) {
583                        returnValue = new Properties();
584                        try (final InputStream stream = new BufferedInputStream(Files.newInputStream(pomPropertiesFile.toPath()))) {
585                          returnValue.load(stream);
586                          beanArchiveRoot = root.toURI().toURL();
587                          root = null;
588                        }
589                      }
590                    } else if (subDirectories.length == 1) {
591                      scanMe = subDirectories[0];
592                    } else {
593                      scanMe = null;
594                    }
595                  }
596
597                } else {
598                  // For common local development purposes, see if
599                  // maven-archiver/pom.properties exists under our
600                  // root; this will give us coordinates for testing
601                  final File mavenArchiverPomProperties = new File(root, "maven-archiver/pom.properties");
602                  if (mavenArchiverPomProperties.exists() && !mavenArchiverPomProperties.isDirectory()) {
603                    returnValue = new Properties();
604                    try (final InputStream stream = new BufferedInputStream(Files.newInputStream(mavenArchiverPomProperties.toPath()))) {
605                      returnValue.load(stream);
606                      beanArchiveRoot = root.toURI().toURL();
607                      root = null;
608                    }
609                  }
610                }
611                if (root != null) {
612                  root = root.getParentFile();
613                }
614              } else if (root != null) {
615                root = root.getParentFile();
616              }
617            }
618          } else {
619            throw new IllegalStateException();
620          }
621
622          if (returnValue == null ||
623              !returnValue.containsKey("groupId") ||
624              !returnValue.containsKey("artifactId") ||
625              !returnValue.containsKey("version")) {
626            returnValue = null;
627          } else {
628            this.coordinates.put(beanArchiveRoot.toURI(), returnValue);
629          }
630        }
631      }
632    }
633    return returnValue;
634  }
