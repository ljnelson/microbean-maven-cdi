060@ApplicationScoped
061public class TestMavenExtension {
062
063  
064  /*
065   * Static fields.
066   */
067
068  
069  /**
070   * The number of instances of this class that have been created (in
071   * the context of JUnit execution; any other usage is undefined).
072   */
073  private static int instanceCount;
074
075
076  /*
077   * Constructors
078   */
079  
080  
081  public TestMavenExtension() {
082    super();
083    instanceCount++;
084  }
085
086
087  /*
088   * Instance methods.
089   */
090
091  private final void onStartup(@Observes @Initialized(ApplicationScoped.class) final Object event,
092                               final RepositorySystem repositorySystem,
093                               final RepositorySystemSession session,
094                               @Resolution final List<RemoteRepository> remoteRepositories,
095                               final TransferListener transferListener)
096    throws DependencyResolutionException {
097    assertNotNull(repositorySystem);
098    assertNotNull(session);
099    assertNotNull(remoteRepositories);
100    assertNotNull(session.getDependencyManager());
101    assertNotNull(transferListener);
102    
103    final CollectRequest collectRequest = new CollectRequest();
104    final Artifact artifact = new DefaultArtifact("org.microbean", "microbean-configuration-cdi", "jar", "0.1.0");
105    collectRequest.setRoot(new Dependency(artifact, JavaScopes.COMPILE));
106    collectRequest.setRepositories(remoteRepositories);
107
108    final DependencyRequest dependencyRequest = new DependencyRequest(collectRequest, DependencyFilterUtils.classpathFilter(JavaScopes.COMPILE));
109    final DependencyResult dependencyResult = repositorySystem.resolveDependencies(session, dependencyRequest);
110    assertNotNull(dependencyResult);
111    final List<ArtifactResult> artifactResults = dependencyResult.getArtifactResults();
112    assertNotNull(artifactResults);
113  }
114  
115  @Test
116  public void testContainerStartup() {
117    final int oldInstanceCount = instanceCount;
118    Main.main(null);
119    assertEquals(oldInstanceCount + 1, instanceCount);
120  }
